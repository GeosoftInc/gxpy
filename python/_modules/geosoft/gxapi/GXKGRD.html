        """

        :param vi:            Flag of input variogram
        :param vo:            Flag of output variogram
        :type  zchan:         str
        :type  in_dat:        GXDAT
        :type  out_grd_dat:   GXDAT
        :type  out_err_dat:   GXDAT
        :type  in_var_name:   str

        :type  vo:            int

        :returns:             0 OK, 1 Error.
